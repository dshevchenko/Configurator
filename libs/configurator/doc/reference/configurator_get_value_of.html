configurator::get_value_of()

Signature

template< typename Option >
std::string configurator::get_value_of();

template< typename Option, typename Value >
Value configurator::get_value_of();

template< typename Option, typename Value >
configurator& configurator::get_value_of( Value& value_of_option );

Description

This function obtains value of option.
Template parameter Option corresponds to type of conceptual option defined with BOOST_CONFIGURATOR_OPTION macro.
Template parameter Value corresponds to type of expected value of option.

Parameters

  1. None.
  2. None.
  3. value_of_option non-const reference to obtaining value of option.

Return

  1. Obtained value of option, std::string type.
  2. Obtained value of option, Value type.
  3. Non-const referense to core class.

Exceptions

std::runtime_error.

Example

BOOST_CONFIGURATOR_OPTION( Host )
BOOST_CONFIGURATOR_OPTION( Port )
BOOST_CONFIGURATOR_OPTION( ExtPort )

int main() {
    boost::cf::configurator conf;
    conf.add< Host >();
    conf.add< Port >();
    conf.add< ExtPort >();
    // ...
    std::string host = conf.get_value_of< Host >();
    // ...
    unsigned int port = conf.get_value_of< Port, unsigned int >();
    // ...
    unsigned long int ext_port = 0;
    conf.get_value_of< ExtPort >( ext_port );
    // ...
}
